№ Циклы while
! При написании скриптов зачастую встаёт задача сделать однотипное действие много раз. Например, вывести товары из списка один за другим. Или просто перебрать все числа от 1 до 10 и для каждого выполнить одинаковый код.
! Для многократного повторения одного участка кода предусмотрены циклы.
! Цикл <span class='ci'>while</span> имеет следующий синтаксис:
@
while (condition) {
    // some code
    // тело цикла
}
@
! Код из тела цикла выполняется, пока условие <span class='ci'>condition</span> истинно.
@
let year = 2016;<br>
while (year <= 2020) {
    console.log(year);
    year = year + 1;
}
@
@
> 2016
> 2017
> 2018
> 2019
> 2020
> 2021
@
! Одно выполнение тела цикла по-научному называется итерация. Цикл в примере выше совершает шесть итераций.
! Если бы строка <span class='ci'>year += 1</span> отсутствовала в примере выше, то цикл бы повторялся (в теории) вечно. 
! Любое выражение или переменная может быть условием цикла, а не только сравнение: условие while вычисляется и преобразуется в логическое значение.
@
let num = 5;
while (num) { // цикл остановится, когда num станет 0
    console.log(num);
    num = num - 1;
}
@
@
5
4
3
2
1
@
! Также можно записывать выражения следующим образом:
@
let num = 5;<br>
num = num - 1; // -> 4
num -= 1;      // -> 3
--num;         // -> 2
@
№ Цикл do...while
! Проверку условия можно разместить под телом цикла, используя специальный синтаксис <span class='ci'>do..while</span>:
@
do {
  // тело цикла
} while (condition);
@
! Цикл сначала выполнит тело, а затем проверит условие condition, и пока его значение равно true, он будет выполняться снова и снова.
! Такая форма синтаксиса оправдана, если вы хотите, чтобы тело цикла выполнилось хотя бы один раз, даже если условие окажется ложным. 
@
let num = 3;
do {
    console.log(num);
    --num;
} while (num);
@
@
> 3
> 2
> 1
> 0
@
